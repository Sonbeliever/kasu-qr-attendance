  <script>
    document.addEventListener("DOMContentLoaded", () => {
      const body = document.body;
      const buttons = Array.from(document.querySelectorAll(".print-mode-btn"));

      function setMode(mode) {
        body.dataset.printMode = mode;
        buttons.forEach((btn) => {
          btn.classList.toggle("is-active", btn.dataset.printMode === mode);
        });
      }

      buttons.forEach((btn) => {
        btn.addEventListener("click", () => setMode(btn.dataset.printMode || "both"));
      });

      setMode("both");
    });
  </script>
